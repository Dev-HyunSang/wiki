# 20231117 - Helix Editor with Golang

# Install Helix Editor

$ brew update && brew upgrade
$ brew install helix

# Setting Golang at Helix Editor

# Install gopls

$ go install golang.org/x/tools/gopls@latest

설치 이후 $PATH 설정이 되어 있지 않는 경우, 실행이 되지 않을 수도 있음.

# ~/.zshrc
export PATH="$HOME/Go/bin:$PATH"

# Setting Themes

  1. CptPotato/helix-themes에서 원하는 테마를 보고 설치할 수 있습니다.
  2. 설치하여 압축 해제 이후 ~/.config/helix/themes에 옮긴 다음에 ~/.config/helix/config.toml에 다음과 같이 설정합니다.
theme = "kanabox_default"
# 혹은
theme = "theme_name"
